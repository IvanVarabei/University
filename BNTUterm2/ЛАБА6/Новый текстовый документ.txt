#include<conio.h>
using namespace std;
void main()
{
	char str[20];
	int f = 0,fl=0;
	const char alph[] = "0123456789abcdefABCDEF";
	cout << "Enter string :";
	gets(str);
	for (int j = 0; str[j] != '\0'; j++)//check for number of commas
	{
		if (str[j] == '.'){fl++;}
	}
	if (fl>1){cout << "no";f++;}
	else
	{
		for (int i = 0; str[i] != '\0'; i++)//check for symbols
		{
			if (str[i] == '.')
			{
				continue;
			}
			if (strchr(alph, str[i]) == NULL)
			{
				cout << "no\n";
				f++;
				break;

			}
		}
	}
	if (!f){cout << "yes";}
	getch();
}
